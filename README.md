# UtilAndroid
**usage:**

    allprojects {
    	repositories {
    		jcenter()
    		maven { url "https://jitpack.io" }
    	}
    }


compile 'com.github.djun100:UtilAndroid:1f5e218a1b180cdaeccdb28f7a5fdc043c50b0f5'

    provided 'com.android.support:support-v4:+'
